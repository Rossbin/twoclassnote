#下载Git，并且安装
---
#在目录下shift + 右键打开powershell
---
    1.git clone GitHub项目的链接

    2.cd .\项目名                //cd到该项目目录下

    3.git config --global user.name "  "

    4.git config --global user.email "  "

    5.git add --all

    6.git commit -m "(在此中说明更新原因)"

    7.git push 
    //第一次使用会弹出输入github用户名 密码的请求
